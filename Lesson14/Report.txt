1.	Для своей базы данных создайте объект для нахождения общего рейтинга топ 50 продавцов за всю историю продаж, 
где рейтинг определяется количеством баллов за количество продаж в день 
(7 и более продаж – 4 балла, 6-5 – 3 балла, 4-3 - 2 балла, 2 и меньше – 1 балл). 
		
		
		DROP VIEW IF EXISTS VTop50Employee

		CREATE VIEW VTop50Employee
		AS
		SELECT TOP 50 WITH TIES T1.EMPSID, T1.FIRSTNAME, T1.LASTNAME
		FROM (
		SELECT R.REQDATE, R.EMPSID, E.FIRSTNAME, E.LASTNAME
				,CASE	WHEN COUNT(R.REQSID)>=7 THEN 4
					WHEN COUNT(R.REQSID)>=5	THEN 3
					WHEN COUNT(R.REQSID)>=3	THEN 2
				ELSE 1 END AS RN	
				--,COUNT(R.REQSID) AS CNT
		FROM [CREDITAPPLICATION].[DBO].[REQUESTS] R
		JOIN [CREDITAPPLICATION].[DBO].[EMPLOYEE] E ON R.EMPSID = E.EMPSID
		GROUP BY R.REQDATE, R.EMPSID, E.FIRSTNAME, E.LASTNAME
		) T1

		GROUP BY T1.EMPSID, T1.FIRSTNAME, T1.LASTNAME
		ORDER BY SUM(T1.RN) DESC

		SELECT * FROM VTop50Employee
